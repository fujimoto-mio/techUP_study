class Patient extends Person{}
   →PatientクラスがPersonクラスを引き継いで利用できるようにextendsでクラスの拡張を実施


private $height = 0;       
private $weight = 0;
    →プロパティの値を設定 
    ※$name, $age, $genderはPersonクラスで設定済


public function __construct($name, $age, $gender, $height, $weight){}
    →$name, $age, $gender, $height, $weightのインスタンスを作成
　　※プロパティの値Patient('山田太郎', 30, 'm', 1.65, 63.5)を受け継いでいる


parent::__construct($name, $age, $gender);
    →Personクラスのコンストラクタを呼び出す


$this -> height = $height;
$this -> weight = $weight;
    →プロパティの初期化
　　※$this -> heightへ$heightを代入、$this -> weightへ$weightを代入


public function calculateStandardWeight(){
     return $this -> height * $this -> height * 22;
 }
    →calculateStandardWeight()へ身長×高さ×22を返す


public function getHeight(){
     return $this -> height;
 }
    →getHeight()へ$this -> heightの値を返す


public function getWeight(){
     return $this -> weight;
 }
    →getWeight()へ$this -> weightの値を返す
